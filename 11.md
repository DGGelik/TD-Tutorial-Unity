using UnityEngine;
using UnityEngine.Events;
using System;

public class GameManager : MonoBehaviour
{
    public static GameManager Instance { get; private set; }

    [Header("Начальные параметры")]
    public int startLives = 20;
    public int initialGold = 200;

    [Header("UI Панели")]
    public GameObject mainMenuPanel;
    public GameObject gameHUD;           // панель с золотом, жизнями, волнами
    public GameObject pausePanel;
    public GameObject gameOverPanel;
    public GameObject victoryPanel;

    public GameState CurrentState { get; private set; } = GameState.MainMenu;

    // Данные игрока
    public int Gold { get; private set; }
    public int Lives { get; private set; }
    public int CurrentWave { get; private set; }
    public int TotalWaves { get; private set; }

    // События
    public UnityEvent<int> OnGoldChanged = new();
    public UnityEvent<int> OnLivesChanged = new();
    public UnityEvent<int> OnWaveChanged = new();
    public UnityEvent<GameState> OnGameStateChanged = new();

    private const string GOLD_KEY = "PlayerGold";

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }

        Instance = this;
        DontDestroyOnLoad(gameObject); // если захочешь в будущем несколько сцен
    }

    private void Start()
    {
        LoadGold();
        ShowMainMenu();
    }

    // ====================== СОСТОЯНИЯ ======================

    public void ShowMainMenu()
    {
        SetState(GameState.MainMenu);
        // Здесь можно добавить анимации появления меню
    }

    public void StartNewGame()
    {
        Lives = startLives;
        CurrentWave = 0;
        
        // Золото берём из сохранений + начальное
        // Gold уже загружен в LoadGold(), можно добавить бонус

        SetState(GameState.Playing);
    }

    public void PauseGame()
    {
        if (CurrentState == GameState.Playing)
        {
            SetState(GameState.Paused);
            Time.timeScale = 0;
        }
    }

    public void ResumeGame()
    {
        if (CurrentState == GameState.Paused)
        {
            SetState(GameState.Playing);
            Time.timeScale = 1f;
        }
    }

    public void GameOver()
    {
        SetState(GameState.GameOver);
        Time.timeScale = 0;
        gameOverPanel?.SetActive(true);
    }

    public void Victory()
    {
        SetState(GameState.Victory);
        Time.timeScale = 0;
        victoryPanel?.SetActive(true);
        SaveGold(); // сохраняем прогресс
    }

    private void SetState(GameState newState)
    {
        CurrentState = newState;
        OnGameStateChanged.Invoke(newState);

        // Управление видимостью панелей
        mainMenuPanel?.SetActive(newState == GameState.MainMenu);
        pausePanel?.SetActive(newState == GameState.Paused);
        gameOverPanel?.SetActive(newState == GameState.GameOver);
        victoryPanel?.SetActive(newState == GameState.Victory);

        if (gameHUD != null)
            gameHUD.SetActive(newState == GameState.Playing);
    }

    // ====================== РЕСУРСЫ ======================

    public void AddGold(int amount)
    {
        Gold += amount;
        OnGoldChanged.Invoke(Gold);
    }

    public bool SpendGold(int amount)
    {
        if (Gold >= amount)
        {
            Gold -= amount;
            OnGoldChanged.Invoke(Gold);
            return true;
        }
        return false;
    }

    public void TakeLife()
    {
        Lives--;
        OnLivesChanged.Invoke(Lives);

        if (Lives <= 0 && CurrentState == GameState.Playing)
            GameOver();
    }

    public void NextWave()
    {
        CurrentWave++;
        OnWaveChanged.Invoke(CurrentWave);
    }

    // ====================== СОХРАНЕНИЕ ======================

    private void LoadGold()
    {
        Gold = PlayerPrefs.GetInt(GOLD_KEY, initialGold);
    }

    private void SaveGold()
    {
        PlayerPrefs.SetInt(GOLD_KEY, Gold);
        PlayerPrefs.Save();
    }

    // Вызывать при выходе из игры
    private void OnApplicationQuit()
    {
        SaveGold();
    }
}

public enum GameState

```csharp
using UnityEngine;

public class EnemyMovement : MonoBehaviour
{
    [SerializeField] private LevelPath path;
    [SerializeField] private float speed = 2f;

    private int currentPointIndex = 0;
    private Vector3[] points;

    private void Start()
    {
        if (path == null || path.points == null || path.points.Length == 0)
        {
            Debug.LogError("LevelPath не назначен или пуст!", this);
            Destroy(gameObject);
            return;
        }

        points = path.points;
        transform.position = points[0];
    }

    private void Update()
    {
        if (currentPointIndex >= points.Length - 1)
        {
            ReachEnd();
            return;
        }

        Vector3 targetPos = points[currentPointIndex + 1];
        Vector3 direction = (targetPos - transform.position).normalized;

        transform.position += direction * speed * Time.deltaTime;

        if (Vector3.Distance(transform.position, targetPos) < 0.1f)
        {
            currentPointIndex++;
            transform.position = targetPos;
        }
    }

    private void ReachEnd()
    {
        GameManager.Instance?.TakeLife();
        Destroy(gameObject);
    }
}
```

#### `TowerController.cs`

```csharp
using UnityEngine;

public class TowerController : MonoBehaviour
{
    [Header("Данные")]
    public TowerData data;

    [Header("Визуал")]
    public Transform turret;
    public Transform firePoint;

    private int currentLevel = 1;
    private float currentDamage;
    private float currentFireRate;
    private float currentRange;

    private Transform target;
    private float fireCountdown;

    private void Start()
    {
        ApplyLevelStats();
    }

    private void ApplyLevelStats()
    {
        currentDamage = data.baseDamage * data.upgradeDamageMultipliers[currentLevel - 1];
        currentFireRate = data.baseFireRate * data.upgradeFireRateMultipliers[currentLevel - 1];
        currentRange = data.baseRange * data.upgradeRangeMultipliers[currentLevel - 1];
    }

    private void Update()
    {
        if (GameManager.Instance.State != GameState.Playing) return;

        UpdateTarget();

        if (target != null)
        {
            LockOnTarget();

            fireCountdown -= Time.deltaTime;
            if (fireCountdown <= 0f)
            {
                Shoot();
                fireCountdown = 1f / currentFireRate;
            }
        }
    }

    private void UpdateTarget()
    {
        Collider2D[] hits = Physics2D.OverlapCircleAll(transform.position, currentRange);
        Transform nearest = null;
        float minDist = float.MaxValue;

        foreach (var hit in hits)
        {
            if (hit.CompareTag("Enemy"))
            {
                float dist = Vector2.Distance(transform.position, hit.transform.position);
                if (dist < minDist)
                {
                    minDist = dist;
                    nearest = hit.transform;
                }
            }
        }

        target = nearest;
    }

    private void LockOnTarget()
    {
        if (turret == null) return;
        Vector3 dir = target.position - turret.position;
        float angle = Mathf.Atan2(dir.y, dir.x) * Mathf.Rad2Deg;
        turret.rotation = Quaternion.Euler(0, 0, angle - 90f);
    }

    private void Shoot()
    {
        if (data.bulletPrefab == null) return;

        GameObject bulletGO = Instantiate(data.bulletPrefab, firePoint.position, firePoint.rotation);
        Bullet bullet = bulletGO.GetComponent<Bullet>();
        if (bullet != null)
        {
            bullet.SetTarget(target, currentDamage);
        }
    }

    public bool Upgrade()
    {
        if (currentLevel >= data.maxLevel) return false;

        int nextLevel = currentLevel + 1;
        int cost = data.upgradeCosts[nextLevel - 2];

        if (GameManager.Instance.SpendGold(cost))
        {
            currentLevel = nextLevel;
            ApplyLevelStats();
            return true;
        }
        return false;
    }

    public void Sell()
    {
        int refund = Mathf.RoundToInt(data.baseCost * 0.65f);
        GameManager.Instance.AddGold(refund);
        Destroy(gameObject);
    }
}
```

#### `BuildManager.cs` (сильно упрощён)

```csharp
using UnityEngine;

public class BuildManager : MonoBehaviour
{
    public static BuildManager Instance { get; private set; }

    [Header("Настройки")]
    [SerializeField] private LayerMask buildableLayerMask;

    // Текущая выбранная башня для постройки
    public TowerData SelectedTowerData { get; private set; }

    // Ghost (призрак) для предпросмотра
    private GameObject ghostTower;
    private SpriteRenderer ghostRenderer;
    private TowerController ghostTowerController;

    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }

        Instance = this;
    }

    private void Update()
    {
        if (GameManager.Instance.CurrentState != GameState.Playing)
        {
            if (ghostTower != null)
                DestroyGhost();
            return;
        }

        HandleGhostTower();
        HandleInput();
    }

    private void HandleGhostTower()
    {
        if (SelectedTowerData == null)
        {
            if (ghostTower != null) DestroyGhost();
            return;
        }

        UpdateGhostPositionAndVisual();
    }

    private void HandleInput()
    {
        // Левая кнопка - поставить башню
        if (Input.GetMouseButtonDown(0) && SelectedTowerData != null)
        {
            TryPlaceTower();
        }

        // Правая кнопка или ESC - отменить выбор
        if (Input.GetMouseButtonDown(1) || Input.GetKeyDown(KeyCode.Escape))
        {
            DeselectTower();
        }
    }

    private void UpdateGhostPositionAndVisual()
    {
        Vector2 mouseWorldPos = Camera.main.ScreenToWorldPoint(Input.mousePosition);

        // Создаём ghost, если его ещё нет
        if (ghostTower == null)
        {
            CreateGhostTower(mouseWorldPos);
        }
        else
        {
            ghostTower.transform.position = mouseWorldPos;
        }

        // Проверка, можно ли строить в этой позиции
        RaycastHit2D hit = Physics2D.Raycast(mouseWorldPos, Vector2.zero, Mathf.Infinity, buildableLayerMask);

        if (hit.collider != null && CanAffordTower())
        {
            ghostRenderer.color = new Color(0.2f, 1f, 0.2f, 0.65f); // Зелёный - можно строить
        }
        else
        {
            ghostRenderer.color = new Color(1f, 0.2f, 0.2f, 0.65f); // Красный - нельзя
        }
    }

    private void CreateGhostTower(Vector2 position)
    {
        ghostTower = Instantiate(SelectedTowerData.turretPrefab, position, Quaternion.identity);
        ghostRenderer = ghostTower.GetComponent<SpriteRenderer>();

        if (ghostRenderer == null)
            ghostRenderer = ghostTower.GetComponentInChildren<SpriteRenderer>();

        // Отключаем функциональность у призрака
        ghostTowerController = ghostTower.GetComponent<TowerController>();
        if (ghostTowerController != null) ghostTowerController.enabled = false;

        Collider2D col = ghostTower.GetComponent<Collider2D>();
        if (col != null) col.enabled = false;

        // Делаем полупрозрачным
        if (ghostRenderer != null)
            ghostRenderer.color = new Color(1, 1, 1, 0.5f);
    }

    private void TryPlaceTower()
    {
        if (SelectedTowerData == null) return;

        Vector2 mouseWorldPos = Camera.main.ScreenToWorldPoint(Input.mousePosition);
        RaycastHit2D hit = Physics2D.Raycast(mouseWorldPos, Vector2.zero, Mathf.Infinity, buildableLayerMask);

        if (hit.collider != null && CanAffordTower())
        {
            if (GameManager.Instance.SpendGold(SelectedTowerData.baseCost))
            {
                // Создаём настоящую башню
                Instantiate(SelectedTowerData.turretPrefab, hit.point, Quaternion.identity);

                Debug.Log($"Башня {SelectedTowerData.towerName} построена!");
                
                // Можно оставить выбор башни (для быстрой постройки нескольких)
                // DeselectTower(); // раскомментируй, если хочешь снимать выбор после постройки
            }
        }
        else
        {
            Debug.Log("Нельзя построить здесь или недостаточно золота!");
        }
    }

    private bool CanAffordTower()
    {
        return SelectedTowerData != null && GameManager.Instance.Gold >= SelectedTowerData.baseCost;
    }

    public void SelectTower(TowerData towerData)
    {
        SelectedTowerData = towerData;
        // Ghost будет создан автоматически в Update
    }

    public void DeselectTower()
    {
        SelectedTowerData = null;
        DestroyGhost();
    }

    private void DestroyGhost()
    {
        if (ghostTower != null)
        {
            Destroy(ghostTower);
            ghostTower = null;
            ghostRenderer = null;
            ghostTowerController = null;
        }
    }

    // Опционально: для UI кнопок
    public void OnTowerButtonClicked(TowerData towerData)
    {
        SelectTower(towerData);
    }
}
```

### 3. `Bullet.cs` (улучшенный)

```csharp
using UnityEngine;

public class Bullet : MonoBehaviour
{
    [SerializeField] private float speed = 12f;
    [SerializeField] private GameObject impactEffect; // опционально

    public float damage { get; private set; }
    private Transform target;

    public void SetTarget(Transform targetTransform, float damageAmount)
    {
        target = targetTransform;
        damage = damageAmount;
    }

    private void Update()
    {
        if (target == null)
        {
            Destroy(gameObject);
            return;
        }

        Vector2 direction = (target.position - transform.position).normalized;
        transform.position += (Vector3)direction * speed * Time.deltaTime;

        // Поворот снаряда
        float angle = Mathf.Atan2(direction.y, direction.x) * Mathf.Rad2Deg;
        transform.rotation = Quaternion.Euler(0, 0, angle - 90f);
    }

    private void OnTriggerEnter2D(Collider2D other)
    {
        if (other.CompareTag("Enemy"))
        {
            EnemyHealth enemyHealth = other.GetComponent<EnemyHealth>();
            if (enemyHealth != null)
            {
                enemyHealth.TakeDamage(damage);
            }

            // Можно добавить эффект попадания
            if (impactEffect != null)
                Instantiate(impactEffect, transform.position, Quaternion.identity);

            Destroy(gameObject);
        }
    }
}
```

```csharp
using UnityEngine;

public class EnemyHealth : MonoBehaviour
{
    [SerializeField] private float maxHealth = 80f;
    [SerializeField] private int goldReward = 15;

    private float currentHealth;
    private EnemyMovement enemyMovement;

    private void Awake()

        currentHealth = maxHealth;

        currentHealth -= amount;

        if (currentHealth <= 0)

    private void Die()
    {
        if (enemyMovement != null)

            enemyMovement.Die();           // через EnemyMovement (там золото и уничтожение)
        }
        else
        {
            GameManager.Instance.AddGold(goldReward);
            Destroy(gameObject);
        }
    }

    public float GetHealthNormalized() => Mathf.Clamp01(currentHealth / maxHealth);
}
```
